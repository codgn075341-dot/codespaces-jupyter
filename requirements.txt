const sqlite3 = require('sqlite3').verbose();
let db = null;

function runAsync(sql, params = []) {
  return new Promise((resolve, reject) => {
    db.run(sql, params, function (err) {
      if (err) return reject(err);
      resolve(this);
    });
  });
}
function getAsync(sql, params = []) {
  return new Promise((resolve, reject) => {
    db.get(sql, params, (err, row) => {
      if (err) return reject(err);
      resolve(row);
    });
  });
}
function allAsync(sql, params = []) {
  return new Promise((resolve, reject) => {
    db.all(sql, params, (err, rows) => {
      if (err) return reject(err);
      resolve(rows);
    });
  });
}

async function init(dbFile = 'data.db') {
  return new Promise((resolve, reject) => {
    db = new sqlite3.Database(dbFile, async (err) => {
      if (err) return reject(err);
      try {
        await runAsync(`CREATE TABLE IF NOT EXISTS users (
          user_id TEXT PRIMARY KEY,
          balance INTEGER NOT NULL DEFAULT 0
        )`);
        await runAsync(`CREATE TABLE IF NOT EXISTS transactions (
          id TEXT PRIMARY KEY,
          user_id TEXT,
          amount INTEGER NOT NULL,
          currency TEXT,
          provider_tx_id TEXT,
          status TEXT,
          intent_id TEXT,
          idempotency_key TEXT,
          provider_response TEXT,
          created_at TEXT
        )`);
        await runAsync(`CREATE TABLE IF NOT EXISTS orders (
          id TEXT PRIMARY KEY,
          user_id TEXT,
          amount INTEGER NOT NULL,
          status TEXT,
          scheduled_at TEXT,
          created_at TEXT
        )`);
        resolve();
      } catch (e) { reject(e); }
    });
  });
}

async function ensureUser(userId) {
  const u = await getAsync(`SELECT * FROM users WHERE user_id = ?`, [userId]);
  if (!u) {
    await runAsync(`INSERT INTO users (user_id, balance) VALUES (?, ?)`, [userId, 0]);
    return { user_id: userId, balance: 0 };
  }
  return u;
}

async function createTransaction(tx) {
  const now = new Date().toISOString();
  await runAsync(
    `INSERT INTO transactions (id, user_id, amount, currency, provider_tx_id, status, intent_id, idempotency_key, provider_response, created_at)
     VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
    [tx.id, tx.user_id, tx.amount, tx.currency || 'thb', tx.provider_tx_id || null, tx.status || 'processing', tx.intent_id || null, tx.idempotency_key || null, tx.provider_response || null, now]
  );
  return await getAsync(`SELECT * FROM transactions WHERE id = ?`, [tx.id]);
}

async function updateTransaction(id, fields = {}) {
  const existing = await getAsync(`SELECT * FROM transactions WHERE id = ?`, [id]);
  if (!existing) throw new Error('tx not found');
  const provider_tx_id = fields.provider_tx_id || existing.provider_tx_id;
  const status = fields.status || existing.status;
  const intent_id = fields.intent_id || existing.intent_id;
  const provider_response = fields.provider_response ? JSON.stringify(fields.provider_response) : existing.provider_response;
  await runAsync(`UPDATE transactions SET provider_tx_id = ?, status = ?, intent_id = ?, provider_response = ? WHERE id = ?`,
    [provider_tx_id, status, intent_id, provider_response, id]);
  return await getAsync(`SELECT * FROM transactions WHERE id = ?`, [id]);
}

async function creditUser(userId, amountSatang) {
  await ensureUser(userId);
  await runAsync(`UPDATE users SET balance = balance + ? WHERE user_id = ?`, [amountSatang, userId]);
  return await getAsync(`SELECT * FROM users WHERE user_id = ?`, [userId]);
}

async function findTransactionByIdempotency(key) {
  return await getAsync(`SELECT * FROM transactions WHERE idempotency_key = ?`, [key]);
}

async function createOrder(id, user_id, amountSatang, scheduledAt = null) {
  const now = new Date().toISOString();
  await runAsync(`INSERT INTO orders (id, user_id, amount, status, scheduled_at, created_at) VALUES (?, ?, ?, ?, ?, ?)`,
    [id, user_id, amountSatang, 'created', scheduledAt, now]);
  return { id, user_id, amount: amountSatang, status: 'created', scheduled_at: scheduledAt, created_at: now };
}

function requireApiKey(req, res, next) {
  if (!REQUIRE_API_KEY) return next(); // allow in test/dev mode
  const auth = (req.headers.authorization || '').replace('Bearer ', '').trim();
  if (!auth || auth !== process.env.API_SERVER_KEY) {
    return res.status(401).json({ success: false, message: 'Unauthorized' });
  }
  next();
}

app.post('/api/secure-charge', requireApiKey, async (req, res) => {
  const { user_id, payment_method, amount, currency = 'thb' } = req.body;
  const idempotency_key = req.body.idempotency_key || req.headers['idempotency-key'] || uuidv4();

  if (!user_id || !payment_method || !amount || Number(amount) <= 0) {
    return res.status(400).json({ success: false, message: 'ข้อมูลไม่ครบถ้วนหรือยอดไม่ถูกต้อง' });
  }

  const amountSatang = Math.round(Number(amount) * 100);
  if (amountSatang < MIN_AMOUNT_SATANG || amountSatang > MAX_AMOUNT_SATANG) {
    return res.status(400).json({ success: false, message: `ยอดต้องอยู่ระหว่าง ${MIN_AMOUNT_THB} - ${MAX_AMOUNT_THB} บาท` });
  }

  const existing = await db.findTransactionByIdempotency(idempotency_key);
  if (existing) {
    return res.status(200).json({ success: true, message: 'duplicate request', transaction: existing });
  }

  const txId = uuidv4();
  await db.createTransaction({
    id: txId,
    user_id,
    amount: amountSatang,
    currency,
    status: 'processing',
    idempotency_key
  });

  try {
    const params = {
      amount: amountSatang,
      currency,
      payment_method: payment_method,
      confirm: true,
      off_session: true
    };
    const stripeOptions = {};
    if (idempotency_key) stripeOptions.idempotencyKey = idempotency_key;

    const intent = await stripe.paymentIntents.create(params, stripeOptions);
    await db.updateTransaction(txId, { intent_id: intent.id, provider_response: intent });

    if (intent.status === 'succeeded' || intent.status === 'requires_capture') {
      await db.updateTransaction(txId, { provider_tx_id: intent.id, status: 'succeeded', provider_response: intent });
      await db.creditUser(user_id, amountSatang);
      const user = await db.getAsync(`SELECT * FROM users WHERE user_id = ?`, [user_id]);
      return res.status(200).json({ success: true, message: 'ตัดเงินสำเร็จ', transactionId: txId, intentStatus: intent.status, newBalance: user.balance });
    }

    if (intent.status === 'requires_action' || intent.status === 'requires_source_action') {
      await db.updateTransaction(txId, { status: 'requires_action', provider_response: intent });
      return res.status(200).json({ success: false, requiresAction: true, clientSecret: intent.client_secret, message: 'ต้องยืนยันการชำระ (3DS)' });
    }

    await db.updateTransaction(txId, { status: 'failed', provider_response: intent });
    return res.status(400).json({ success: false, message: 'ชำระเงินไม่สำเร็จ', intentStatus: intent.status });
  } catch (err) {
    console.error('Charge error', err && (err.raw || err));
    await db.updateTransaction(txId, { status: 'failed', provider_response: err && err.raw ? err.raw : { message: err.message } });
    return res.status(500).json({ success: false, message: 'เกิดข้อผิดพลาดขณะตัดเงิน (ดู log)' });
  }
});

// Webhook endpoint (Stripe)
app.post('/webhook', express.raw({ type: 'application/json' }), (req, res) => {
  const sig = req.headers['stripe-signature'] || '';
  const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET;
  let event;
  try {
    event = stripe.webhooks.constructEvent(req.body, sig, webhookSecret);
  } catch (err) {
    console.error('Webhook signature verification failed', err.message);
    return res.status(400).send(`Webhook Error: ${err.message}`);
  }

  (async () => {
    try {
      if (event.type === 'payment_intent.succeeded') {
        const intent = event.data.object;
        const tx = await db.getAsync(`SELECT * FROM transactions WHERE intent_id = ?`, [intent.id]);
        if (tx && tx.status !== 'succeeded') {
          await db.updateTransaction(tx.id, { status: 'succeeded', provider_tx_id: intent.id, provider_response: intent });
          await db.creditUser(tx.user_id, intent.amount);
        }
      } else if (event.type === 'payment_intent.payment_failed') {
        const intent = event.data.object;
        const tx = await db.getAsync(`SELECT * FROM transactions WHERE intent_id = ?`, [intent.id]);
        if (tx) {
          await db.updateTransaction(tx.id, { status: 'failed', provider_response: intent.last_payment_error || intent });
        }
      }
      res.json({ received: true });
    } catch (e) {
      console.error('Webhook processing error', e);
      res.status(500).send('internal error');
    }
  })();
});

(async function(){
  const startBtn = document.getElementById('start-btn');

  const openLinkBtn = document.getElementById('open-link');

  async function getCsrf(){
    const r = await fetch('/csrf-token', { credentials: 'same-origin' });
    if (!r.ok) throw new Error('ไม่สามารถขอ CSRF token');
    const j = await r.json();
    return j.csrfToken;
  }

  startBtn.addEventListener('click', async () => {
    try {
      startBtn.disabled = true;
      startBtn.textContent = 'กำลังสร้าง session...';
      const csrf = await getCsrf();
      const user_id = userIdInput.value || 'user-unknown';
      const resp = await fetch('/consent/init', {
        method: 'POST',
        credentials: 'same-origin',
        headers: {'Content-Type':'application/json','CSRF-Token':csrf},
        body: JSON.stringify({ user_id })
      });
      const j = await resp.json();
      if (!j.success) { showError(j.message || 'สร้าง session ล้มเหลว'); startBtn.disabled=false; startBtn.textContent='เริ่มการยินยอม (Request Consent)'; return; }

      redirectUrlEl.textContent = j.redirectUrl;
      sessionIdEl.textContent = j.sessionId;
      result.style.display = 'block';

      startBtn.disabled = false;
      startBtn.textContent = 'เริ่มการยินยอม (Request Consent)';
    }
  });

  openLinkBtn.addEventListener('click', () => {
    const url = redirectUrlEl.textContent;
    if (url && url !== '—') window.open(url, '_blank');
  });

  checkStatusBtn.addEventListener('click', async () => {
    try {
      const session = sessionIdEl.textContent;
      if (!session || session === '—') return showError('ไม่มี session id');
      const r = await fetch('/mandate/status?session=' + encodeURIComponent(session), { credentials: 'same-origin' });
      const j = await r.json();
      if (!j.success) return showError(j.message || 'ไม่สามารถตรวจสอบสถานะได้');
      mandateStatusEl.textContent = (j.mandate && j.mandate.status) ? j.mandate.status : 'not found';
    } catch (e) {
      console.error(e);
      showError('ตรวจสอบสถานะล้มเหลว');
    }
  });
})();

# Secure-charge demo (Stripe test) — Complete bundle

เตือน: โค้ดนี้เป็นโซลูชันที่ใช้งานได้จริง แต่ค่าเริ่มต้นใช้ Stripe test keys (sandbox). ห้ามใช้กับข้อมูลจริงจนกว่าจะทดสอบและมีการอนุญาต/รีวิวความปลอดภัยเรียบร้อย.

1) วางไฟล์ทั้งหมดในโฟลเดอร์โปรเจคเดียวกัน (รวมโฟลเดอร์ `public/` ด้วย)
2) คัดลอก `.env.example` เป็น `.env` และแก้:
   - STRIPE_SECRET_KEY=sk_test_xxx
   - STRIPE_PUBLISHABLE_KEY=pk_test_xxx
   - STRIPE_WEBHOOK_SECRET=whsec_xxx (จาก stripe listen)
   - API_SERVER_KEY=some_secret (ถ้าใช้งาน production ให้ตั้ง REQUIRE_API_KEY=true)
   - MIN_AMOUNT_THB / MAX_AMOUNT_THB ตามต้องการ (ค่าปัจจุบัน 60000-300000)

3) ติดตั้ง dependency และรัน:
   npm install
   npm start

4) ตั้งค่า webhook local (แนะนำใช้ Stripe CLI):
   stripe login
   stripe listen --forward-to localhost:5000/webhook
   - คัดลอก webhook signing secret ที่ได้มาใส่ใน .env (STRIPE_WEBHOOK_SECRET)

5) เปิดหน้า checkout:
   http://localhost:5000/checkout.html
   - ทดสอบการ์ด (Stripe test):
     - สำเร็จทันที: 4242 4242 4242 4242
     - 3DS required: 4000 0025 0000 3155

6) ตรวจดูฐานข้อมูล/รายการ:
   - ถ้าตั้ง REQUIRE_API_KEY=false (dev) คุณสามารถเรียก:
     http://localhost:5000/admin/transactions
     http://localhost:5000/admin/users
   - หาก REQUIRE_API_KEY=true ให้เรียกด้วย header Authorization: Bearer <API_SERVER_KEY>

- ใช้ publishable key ใน client เท่านั้น; secret key อยู่ที่ server (.env)
